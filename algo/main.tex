	\begin{algorithm}
		\caption{Genetic VANS}
		\label{array-sum}
		\begin{algorithmic}[1]
    	\State \mathbb{input: }\texttt{initial circuit }$\mathcal{C}(\vec{\theta}$)\textt{, alphabet of blocks }$\mathcal{A }  $\texttt{desired accuracy }$\epsilon$
		 \State \texttt{converged} $\leftarrow$ False
			\State set $E^{*} = E[\mathcal{C}(\vec{\theta^{*}})]$
			\While{ not \texttt{converged}}
			\State \texttt{pick a block $B_i (\tilde{\theta})$ from $\mathcal{A} $ with prob $p_i$, \texttt{and randomly place it in } $\mathcal{C}$.
			\State \underline{initialize} $\mathcal{C}_{B_{i}}(\vec{\theta} \sim \vec{\theta^{*}}, \tilde{\theta} \sim \epsilon})$\texttt{ and compute }$\hat{E} = E[\mathcal{C}_{B_i}(\vec{\theta^{*}}
			, \tilde{\theta}^{*})]$
			\State \textbf{if} $\hat{E} < E^{*}$:
			\State $\;\;\;\;\;$ \texttt{accept $B_i$} (now $\mathcal{C}$ has block $B_i$ at the corresponding position)
			\State $\;\;\;\;\;$ set $\vec{\theta} \leftarrow (\vec{\theta^{*}}
			, \tilde{\theta}^{*})$
			\State $\;\;\;\;\;$ \texttt{suppress 1-qubit unitaries if possible}
            \State $\;\;\;\;\;$ \texttt{suppress consecutive CNOTs/1-qubit unitaries if possible}
    	\State \textbf{if} $E^{*}-E_g < \epsilon$:
    		\State $\;\;\;\;\;$  \texttt{converged} $\leftarrow$ False
    		\State $\;\;\;\;\;$ \texttt{return $$}
		\end{algorithmic}

	\State
	\end{algorithm}

	\subsection*{Note:}
    $ E[\mathcal{C}(\vec{\theta^{*}})$ is obtained from classical optimization routine, e.g. Adam.
